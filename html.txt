        function generateSearchURL() {
            var fname = $.trim($('#fname').val()) != '' ? $("#fname").val().trim() : '';
            var mname = $.trim($('#mname').val()) != '' ? $("#mname").val().trim() : '';
            var lname = $.trim($('#lname').val()) != '' ? $("#lname").val().trim() : '';
            var state = $.trim($('#statesDDL').val()) != '' ? $("#statesDDL").val().trim() : '';
            var city = $.trim($('#city').val()) != '' ? $("#city").val().trim() : '';
            var party = $.trim($('#partyDDL').val()) != '' ? '-' + $("#partyDDL").val().replace(/\ /g, '+').trim() : '';
            var year = $.trim($('#yearBornDDL').val()) != '' ? '-' + $("#yearBornDDL").val().replace(/\ /g, '+').replace('-', 'to').trim() : '';
            var loc = '';
            if (city.length > 0 && state.length > 0) {
                loc = city.replace(/\ /g, "+") + '-' + state.replace(/\ /g, "+");
            }
            else if (city.length > 0) {
                loc = city.replace(/\ /g, "+") + '-' + "us";
            }
            else if (state.length > 0) {
                loc = state.replace(/\ /g, "+");
            }
            var name = '';
            name = fname.length > 0 ? fname + ' ' : '';
            name += mname.length > 0 ? mname + ' ' : '';
            name += lname.length > 0 ? lname : '';
            name = name.replace(/\ /g, '+').trim();
            var url;
            if (loc.length == 0 && name.length == 0 && party.length == 0 && year.length == 0) {
                url = ('/voters/1').replace('/-', '/').toLowerCase();
            }
            else if (name.length > 0 && loc.length == 0 && party.length == 0 && year.length == 0) {
                url = ('/voters/' + name.replace(/\+/g, '-') + '/1').replace('/-', '/').toLowerCase();
            }
            else if (loc.length > 0 && name.length == 0 && party.length == 0 && year.length == 0) {
                url = ('/voters/' + loc + '/1').replace('/-', '/').toLowerCase();
            }
            else {
                loc = loc.length == 0 ? 'us' : loc;
                url = ('/voters/' + loc + '/' + name + party + year + '/1').replace('/-', '/').toLowerCase();
            }
            return url;
        }
